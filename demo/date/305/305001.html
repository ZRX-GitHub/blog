创建表


/*
create table table_name ( 参数 ) ： 参数和参数之间使用 逗号隔开
参数： 列名 类型
*/
-- 创建一个用户表： id ( 主键：唯一非空 ) 、account 、password 、create_time 、height 、phone

-- 创建表
create table tb_user(
	   id int primary key,
	   account varchar2(6),
	   password varchar2(8),
	   height number(3,2),
	   phone char(11),
	   birth date,
	   create_time timestamp 
);

-- 插入全部数据： 完整插入插入所有字段
insert into tb_user values 
(1,'admin','123456',1.99,'15899998888',to_date('1990-10-10','yyyy-mm-dd'),sysdate);
commit;

insert into tb_user values 
(2,'admin','123456',1.99,'1589999',to_date('1990-10-10','yyyy-mm-dd'),sysdate);
commit;

-- 插入部分字段
insert into tb_user 
(id,account,password,height,phone,create_time) 
values 
(3,'user','user',1.88,'131312312',sysdate);
commit;

-- 验证可变长度字符串
select length(password),length(phone) from tb_user;

 

修改表

alter table tb_user add  addr varchar2(100);  -- 添加 新的 属性列
alter table tb_user modify height number(5,2);  -- 修改 属性列的属性定义
alter table tb_user drop column addr;  -- 删除 属性列
alter table tb_user rename addr to addr_new;  -- 重命名 属性列 的名称

 

删除表

drop table tb_user;

 

修改属性列里的属性

根据主键修改，不建议根据其他属性修改，不建议范围修改。

update tb_user set phone = '888' where id = 2;
commit;

 

删除属性列

根据主键删除，不建议根据其他属性删除，不建议范围删除。

delete tb_user where id = 3;
commit;
